组件

<body>
  <div id="app">
    <important-common></important-common>
  </div>
  <script type="text/javascript" src="../js/1.js"></script>
</body>

// 全局
Vue.component('important-common', {
    template: '<h2 style="color:red">我是Template模板, id是important-common</h2>'
})

或者局部：

var vm = new Vue({
    el: '#app',
    components: {
        'important-common': {
            template: '<h2 style="color:red">我是Template模板, id是important-common 2</h2>'
        }
    }
});

上面是两种组件模板template的写法，还可以把模板写在body中，然后在new Vue({})实例对象中引用：

<body>
    <template id="important-common">
        <h2 style="color:red">我是Template模板, id是important-common</h2>
    </template>

    <div id="app">
      <important-common></important-common>
    </div>
    <script type="text/javascript" src="../js/1.js"></script>
</body>

var vm = new Vue({
    el: '#app',
    template: '#important-common' # 给vue注册，不可省略
    // template: '#important-common-2'
    // ...
    // components: {
    //     'important-common': {
    //         template: '<h2 style="color:red">我是Template模板, id是important-common 2</h2>'
    //     }
    // }
});

组件之属性Prop

<body>
<div id="app">
    <child message="hello!"></child>
</div>

<script>
// 注册
Vue.component('child', {
  // 声明 props
  props: ['message'],
  // 同样也可以在 vm 实例中像 “this.message” 这样使用
  template: '<span>{{ message }}</span>'
})
// 创建根实例
new Vue({
  el: '#app'
})
</script>
</body>

组件Prop之动态属性

<body>
<div id="app">
    <div>
      <input v-model="parentMsg">
      <br>
      <child v-bind:message="parentMsg"></child>
    </div>
</div>

<script>
// 注册
Vue.component('child', {
  // 声明 props
  props: ['message'],
  // 同样也可以在 vm 实例中像 “this.message” 这样使用
  template: '<span>{{ message }}</span>'
})
// 创建根实例
new Vue({
  el: '#app',
  data: {
    parentMsg: '父组件内容'
  }
})
</script>
</body>

自定义事件

<body>
  <div id="app">
      <p>{{ total }}</p>
      <button-counter v-on:increment="incrementTotal"></button-counter>
      <button-counter v-on:increment="incrementTotal"></button-counter>
  </div>
  <script type="text/javascript" src="../js/1.js"></script>
</body>

Vue.component('button-counter', {
    template: '<button v-on:click="incrementHandler">{{ counter }}</button>',
    data() {
        return {
            counter: 0,
        }
    },
    methods: {
        incrementHandler: function() {
            this.counter += 1;
            this.$emit('increment');
        }
    },
})

var vm = new Vue({
    el: '#app',
    data: {
        total: 0,
    },
    methods: {
        incrementTotal: function() {
            this.total += 1;
        }
    },
});
